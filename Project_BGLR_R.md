# --- Extract data ---
y_true <- Y         # Actual yield values
y_pred <- fit$yHat  # Predicted yield (posterior mean)

# --- 1. Metric functions ---
r2 <- function(y, yhat) {
  ss_res <- sum((y - yhat)^2)
  ss_tot <- sum((y - mean(y))^2)
  1 - ss_res / ss_tot
}

rmse <- function(y, yhat) sqrt(mean((y - yhat)^2))
rmspe <- function(y, yhat) sqrt(mean(((y - yhat) / y)^2, na.rm = TRUE)) * 100

# --- 2. Compute base metrics ---
metrics <- data.frame(
  Metric = c("R²", "RMSE", "RMSPE (%)"),
  Value = c(r2(y_true, y_pred),
            rmse(y_true, y_pred),
            rmspe(y_true, y_pred))
)

# --- 3. Bootstrap confidence intervals (robust version) ---
set.seed(42)
n_boot <- 1000
n <- length(y_true)

boot_ci <- function(metric_fun, y, yhat, n_boot = 1000, ci = 0.95) {
  vals <- numeric(n_boot)
  for (i in seq_len(n_boot)) {
    idx <- sample(seq_along(y), n, replace = TRUE)
    val <- suppressWarnings(try(metric_fun(y[idx], yhat[idx]), silent = TRUE))
    if (is.numeric(val) && length(val) == 1 && is.finite(val)) vals[i] <- val else vals[i] <- NA
  }
  vals <- vals[is.finite(vals)]
  if (length(vals) < 5) {
    return(data.frame(Mean = NA, Lower = NA, Upper = NA))
  }
  ci_vals <- quantile(vals, probs = c((1 - ci) / 2, 1 - (1 - ci) / 2), na.rm = TRUE)
  data.frame(
    Mean = mean(vals, na.rm = TRUE),
    Lower = ci_vals[1],
    Upper = ci_vals[2]
  )
}

r2_ci <- boot_ci(r2, y_true, y_pred)
rmse_ci <- boot_ci(rmse, y_true, y_pred)
rmspe_ci <- boot_ci(rmspe, y_true, y_pred)

ci_df <- data.frame(
  Metric = c("R²", "RMSE", "RMSPE (%)"),
  Mean = c(r2_ci$Mean, rmse_ci$Mean, rmspe_ci$Mean),
  Lower = c(r2_ci$Lower, rmse_ci$Lower, rmspe_ci$Lower),
  Upper = c(r2_ci$Upper, rmse_ci$Upper, rmspe_ci$Upper)
)

print(ci_df)
